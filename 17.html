import csv
def add_emp():
  with open('emp.csv', 'a', newline='') as f:
    W=csv.writer(f)
    name = input("Enter Employee Name:")
    empno = int(input("Enter Employee Number:"))
    salary = int(input("Enter Employee Salary:"))
    L=[empno, name, salary]
    W.writerow(L)

def display_sal():
  f = open('emp.csv','r')
  R=csv.reader(f)
  for i in R:
    if int(i[2])>10000:
      print(i)

  f.close()


while True:
  print("1.Add employee/n2.Display all employees with salary above 10000.\n3.Exit\n")
  ch = int(input("Enter choice:"))
  if ch==1:
    add_emp()
  elif ch==2:
    display_sal()
  elif: ch==3:
    break
  else:
    print("invalid choice")
