Q:Write a program to read content in a text file story.txt and count number of uppercase, lowercase, special characters and digits in it

def count_chars():
    upper = 0
    lower = 0
    digits = 0
    special = 0
    
    f = open('story.txt', 'r')
    text = f.read()
    
    for i in text:
        if i.isupper():
            upper += 1
        elif i.islower():
            lower += 1
        elif i.isdigit():
            digits += 1
        elif not i.isspace():
            special += 1
    f.close()

    print("Uppercase letters:", upper)
    print("Lowercase letters:", lower)
    print("Digits:", digits)
    print("Special characters:", special)

count_chars()
