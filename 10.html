Q:Write a program to read content of a text file story.txt and print those words which starts with S.

def print_words():
    f = open('story.txt', 'r')
    text = f.read()
    words = text.split()
    for w in words:
        if w.startswith('S') or w.startswith('s'):
            print(w)
    f.close()


print_words()
