Q:Write a program to find average of elements in a list using function?

def average(l):
    total_sum = 0
    count = len(l)
    
    for i in l:
        total_sum += i
    
    average = total_sum / count
    
    return average

LI=eval(input("Enter a list of numbers:"))
a = average(LI)
print("The average is:", a)
